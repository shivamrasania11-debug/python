declare
str varchar2(10):='&str';
rev varchar2(10);
len varchar2(20);
begin
len:=length(str);
loop
	rev:=rev||substr(str,len,1);
	len:=len-1;
	exit when len=0;
end loop;
	if rev = str then
	dbms_output.put_line('string is palindrome');
	else
	dbms_output.put_line('string is not palindrome');
	end if;
end;
