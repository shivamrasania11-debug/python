declare
n number(5):=&n;
sum number(5):=0;
r number(5);
n1 number(5);
begin
n1:=n;
while n>0 loop
		r:=MOD(n,10);
		sum:=sum+power(r,3);
		n:=TRUNC(n/10);
	end loop;
if sum=n1 then
	dbms_output.put_line('number is armstrong');
else
	dbms_output.put_line('number is not armstrong');
end if;
end;
