declare
n number(5):=&n;
rev number(5):=0;
r number(5):=0;
n1 number(5);
begin
n1:=n;
while n>0 
	loop
		r:=n mod 10;
		rev:=(rev * 10)+r;
		n:=n/10;
	end loop;
if rev=n1 then
	dbms_output.put_line('number is palindrome');
else
	dbms_output.put_line('number is not palindrome');
end if;
end;
